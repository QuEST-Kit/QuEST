74 void getZYZRotAnglesFromComplexPair(Complex alpha, Complex beta, qreal* rz2, qreal* ry, qreal* rz1) {
75 
76  qreal alphaMag = sqrt(alpha.real*alpha.real + alpha.imag*alpha.imag);

86 void getComplexPairAndPhaseFromUnitary(ComplexMatrix2 u, Complex* alpha, Complex* beta, qreal* globalPhase) {
87 
88  qreal r0c0Phase = atan2(u.r0c0.imag, u.r0c0.real);
89  qreal r1c1Phase = atan2(u.r1c1.imag, u.r1c1.real);
90  *globalPhase = (r0c0Phase + r1c1Phase)/2.0;
91 
92  qreal cosPhase = cos(*globalPhase);
93  qreal sinPhase = sin(*globalPhase);
94  alpha->real = u.r0c0.real*cosPhase + u.r0c0.imag*sinPhase;
95  alpha->imag = u.r0c0.imag*cosPhase - u.r0c0.real*sinPhase;
96  beta->real = u.r1c0.real*cosPhase + u.r1c0.imag*sinPhase;
97  beta->imag = u.r1c0.imag*cosPhase - u.r1c0.real*sinPhase;
98 }

105 int generateMeasurementOutcome(qreal zeroProb, qreal *outcomeProb) {
106 
107  // randomly choose outcome
108  int outcome;
109  if (zeroProb < REAL_EPS)
110  outcome = 1;
111  else if (1-zeroProb < REAL_EPS)
112  outcome = 0;
113  else
114  outcome = (genrand_real1() > zeroProb);
115 
116  // set probability of outcome
117  if (outcome == 0)
118  *outcomeProb = zeroProb;
119  else
120  *outcomeProb = 1 - zeroProb;
121 
122  return outcome;
123 }

135 void seedQuESTDefault(){
136  // init MT random number generator with three keys -- time, pid and a hash of hostname
137  // for the MPI version, it is ok that all procs will get the same seed as random numbers will only be
138  // used by the master process
139 
140  struct timeval tv;
141  gettimeofday(&tv, NULL);
142 
143  double time_in_mill =
144  (tv.tv_sec) * 1000 + (tv.tv_usec) / 1000 ; // convert tv_sec & tv_usec to millisecond
145 
146  unsigned long int pid = getpid();
147  unsigned long int msecs = (unsigned long int) time_in_mill;
148  char hostName[MAXHOSTNAMELEN+1];
149  gethostname(hostName, sizeof(hostName));
150  unsigned long int hostNameInt = hashString(hostName);
151 
152  unsigned long int key[3];
153  key[0] = msecs; key[1] = pid; key[2] = hostNameInt;
154  init_by_array(key, 3);
155 }
