790 void densmatr_collapseToKnownProbOutcome(Qureg qureg, const int measureQubit, int outcome, qreal totalStateProb) {
791 
792  // only (global) indices (as bit sequence): '* outcome *(n+q) outcome *q are spared
793  // where n = measureQubit, q = qureg.numQubitsRepresented.
794  // We can thus step in blocks of 2^q+n, killing every second, and inside the others,
795  // stepping in sub-blocks of 2^q, killing every second.
796  // When outcome=1, we offset the start of these blocks by their size.
797  long long int innerBlockSize = (1LL << measureQubit);
798  long long int outerBlockSize = (1LL << (measureQubit + qureg.numQubitsRepresented));
799 
800  // Because there are 2^a number of nodes(/chunks), each node will contain 2^b number of blocks,
801  // or each block will span 2^c number of nodes. Similarly for the innerblocks.
802  long long int locNumAmps = qureg.numAmpsPerChunk;
803  long long int globalStartInd = qureg.chunkId * locNumAmps;
804  int innerBit = extractBit(measureQubit, globalStartInd);
805  int outerBit = extractBit(measureQubit + qureg.numQubitsRepresented, globalStartInd);
806 
807  // If this chunk's amps are entirely inside an outer block
808  if (locNumAmps <= outerBlockSize) {
809 
810  // if this is an undesired outer block, kill all elems
811  if (outerBit != outcome)
812  return zeroSomeAmps(qureg, 0, qureg.numAmpsPerChunk);
813 
814  // othwerwise, if this is a desired outer block, and also entirely an inner block
815  if (locNumAmps <= innerBlockSize) {
816 
817  // and that inner block is undesired, kill all elems
818  if (innerBit != outcome)
819  return zeroSomeAmps(qureg, 0, qureg.numAmpsPerChunk);
820  // otherwise normalise all elems
821  else
822  return normaliseSomeAmps(qureg, totalStateProb, 0, qureg.numAmpsPerChunk);
823  }
824 
825  // otherwise this is a desired outer block which contains 2^a inner blocks; kill/renorm every second inner block
826  return alternateNormZeroingSomeAmpBlocks(
827  qureg, totalStateProb, innerBit==outcome, 0, qureg.numAmpsPerChunk, innerBlockSize);
828  }
829 
830  // Otherwise, this chunk's amps contain multiple outer blocks (and hence multiple inner blocks)
831  long long int numOuterDoubleBlocks = locNumAmps / (2*outerBlockSize);
832  long long int firstBlockInd;
833 
834  // alternate norming* and zeroing the outer blocks (with order based on the desired outcome)
835  // These loops aren't parallelised, since they could have 1 or 2 iterations and will prevent
836  // inner parallelisation
837  if (outerBit == outcome) {
838 
839  for (long long int outerDubBlockInd = 0; outerDubBlockInd < numOuterDoubleBlocks; outerDubBlockInd++) {
840  firstBlockInd = outerDubBlockInd*2*outerBlockSize;
841 
842  // *norm only the desired inner blocks in the desired outer block
843  alternateNormZeroingSomeAmpBlocks(
844  qureg, totalStateProb, innerBit==outcome,
845  firstBlockInd, outerBlockSize, innerBlockSize);
846 
847  // zero the undesired outer block
848  zeroSomeAmps(qureg, firstBlockInd + outerBlockSize, outerBlockSize);
849  }
850 
851  } else {
852 
853  for (long long int outerDubBlockInd = 0; outerDubBlockInd < numOuterDoubleBlocks; outerDubBlockInd++) {
854  firstBlockInd = outerDubBlockInd*2*outerBlockSize;
855 
856  // same thing but undesired outer blocks come first
857  zeroSomeAmps(qureg, firstBlockInd, outerBlockSize);
858  alternateNormZeroingSomeAmpBlocks(
859  qureg, totalStateProb, innerBit==outcome,
860  firstBlockInd + outerBlockSize, outerBlockSize, innerBlockSize);
861  }
862  }
