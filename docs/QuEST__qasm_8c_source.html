55 void qasm_setup(Qureg* qureg) {
56 
57  // populate and attach QASM logger
58  QASMLogger *qasmLog = malloc(sizeof *qasmLog);
59  qureg->qasmLog = qasmLog;
60  if (qasmLog == NULL)
61  bufferOverflow();
62 
63  qasmLog->isLogging = 0;
64  qasmLog->bufferSize = BUF_INIT_SIZE;
65  qasmLog->buffer = malloc(qasmLog->bufferSize * sizeof *(qasmLog->buffer));
66  if (qasmLog->buffer == NULL)
67  bufferOverflow();
68 
69  // add headers and quantum / classical register creation
70  qasmLog->bufferFill = snprintf(
71  qasmLog->buffer, qasmLog->bufferSize,
72  "OPENQASM 2.0;\nqreg %s[%d];\ncreg %s[%d];\n",
73  QUREG_LABEL, qureg->numQubitsRepresented,
74  MESREG_LABEL, qureg->numQubitsRepresented);
75  if (qasmLog->bufferFill >= qasmLog->bufferSize)
76  bufferOverflow();
77 }

87 void addStringToQASM(Qureg qureg, char line[], int lineLen) {
88 
89  char* buf = qureg.qasmLog->buffer;
90  int bufSize = qureg.qasmLog->bufferSize;
91  int bufFill = qureg.qasmLog->bufferFill;
92 
93  // grow QASM buffer if necessary
94  if (lineLen + bufFill > bufSize) {
95 
96  int newBufSize = BUF_GROW_FAC * bufSize;
97  if (lineLen + bufFill > newBufSize)
98  bufferOverflow();
99 
100  char* newBuffer = malloc(newBufSize * sizeof *newBuffer);
101  sprintf(newBuffer, "%s", buf);
102  free(buf);
103 
104  qureg.qasmLog->bufferSize = newBufSize;
105  qureg.qasmLog->buffer = newBuffer;
106  bufSize = newBufSize;
107  buf = newBuffer;
108  }
109 
110  // add new str
111  int addedChars = snprintf(buf+bufFill, bufSize-bufFill, "%s", line);
112  qureg.qasmLog->bufferFill += addedChars;
113 }

125 void addGateToQASM(Qureg qureg, TargetGate gate, int* controlQubits, int numControlQubits, int targetQubit, qreal* params, int numParams) {
126 
127  int len = 0;
128  char line[MAX_LINE_LEN + 1]; // for trailing \0
129 
130  // add control labels
131  for (int i=0; i < numControlQubits; i++)
132  len += snprintf(line+len, MAX_LINE_LEN-len, "%s", CTRL_LABEL_PREF);
133 
134  // add target gate
135  len += snprintf(line+len, MAX_LINE_LEN-len, "%s", qasmGateLabels[gate]);
136 
137  // add parameters
138  if (numParams > 0) {
139  len += snprintf(line+len, MAX_LINE_LEN-len, "(");
140  for (int i=0; i < numParams; i++) {
141  len += snprintf(line+len, MAX_LINE_LEN-len, REAL_QASM_FORMAT, params[i]);
142  if (i != numParams - 1)
143  len += snprintf(line+len, MAX_LINE_LEN-len, ",");
144  }
145  len += snprintf(line+len, MAX_LINE_LEN-len, ")");
146  }
147 
148  // add space
149  len += snprintf(line+len, MAX_LINE_LEN-len, " ");
150 
151  // add control qubits
152  for (int i=0; i < numControlQubits; i++)
153  len += snprintf(line+len, MAX_LINE_LEN-len, "%s[%d],", QUREG_LABEL, controlQubits[i]);
154 
155  // add target qubit, colon and newline
156  len += snprintf(line+len, MAX_LINE_LEN-len, "%s[%d];\n", QUREG_LABEL, targetQubit);
157 
158  // check whether we overflowed buffer
159  if (len >= MAX_LINE_LEN)
160  bufferOverflow();
161 
162  addStringToQASM(qureg, line, len);
163 }
