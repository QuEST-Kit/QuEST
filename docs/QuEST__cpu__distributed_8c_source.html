52 qreal densmatr_calcTotalProb(Qureg qureg) {
53 
54  // computes the trace by summing every element ("diag") with global index (2^n + 1)i for i in [0, 2^n-1]
55 
56  // computes first local index containing a diagonal element
57  long long int diagSpacing = 1LL + (1LL << qureg.numQubitsRepresented);
58  long long int numPrevDiags = (qureg.chunkId>0)? 1+(qureg.chunkId*qureg.numAmpsPerChunk)/diagSpacing : 0;
59  long long int globalIndNextDiag = diagSpacing * numPrevDiags;
60  long long int localIndNextDiag = globalIndNextDiag % qureg.numAmpsPerChunk;
61  long long int index;
62 
63  qreal rankTotal = 0;
64  qreal y, t, c;
65  c = 0;
66 
67  // iterates every local diagonal
68  for (index=localIndNextDiag; index < qureg.numAmpsPerChunk; index += diagSpacing) {
69 
70  // Kahan summation - brackets are important
71  y = qureg.stateVec.real[index] - c;
72  t = rankTotal + y;
73  c = ( t - rankTotal ) - y;
74  rankTotal = t;
75  }
76 
77  // combine each node's sum of diagonals
78  qreal globalTotal;
79  if (qureg.numChunks > 1)
80  MPI_Allreduce(&rankTotal, &globalTotal, 1, MPI_QuEST_REAL, MPI_SUM, MPI_COMM_WORLD);
81  else
82  globalTotal = rankTotal;
83 
84  return globalTotal;
85 }

87 qreal statevec_calcTotalProb(Qureg qureg){
88  // Implemented using Kahan summation for greater accuracy at a slight floating
89  // point operation overhead. For more details see https://en.wikipedia.org/wiki/Kahan_summation_algorithm
90  qreal pTotal=0;
91  qreal y, t, c;
92  qreal allRankTotals=0;
93  long long int index;
94  long long int numAmpsPerRank = qureg.numAmpsPerChunk;
95  c = 0.0;
96  for (index=0; index<numAmpsPerRank; index++){
97  // Perform pTotal+=qureg.stateVec.real[index]*qureg.stateVec.real[index]; by Kahan
98  y = qureg.stateVec.real[index]*qureg.stateVec.real[index] - c;
99  t = pTotal + y;
100  // Don't change the bracketing on the following line
101  c = ( t - pTotal ) - y;
102  pTotal = t;
103  // Perform pTotal+=qureg.stateVec.imag[index]*qureg.stateVec.imag[index]; by Kahan
104  y = qureg.stateVec.imag[index]*qureg.stateVec.imag[index] - c;
105  t = pTotal + y;
106  // Don't change the bracketing on the following line
107  c = ( t - pTotal ) - y;
108  pTotal = t;
109  }
110  if (qureg.numChunks>1)
111  MPI_Allreduce(&pTotal, &allRankTotals, 1, MPI_QuEST_REAL, MPI_SUM, MPI_COMM_WORLD);
112  else
113  allRankTotals=pTotal;
114 
115  return allRankTotals;

368 void copyVecIntoMatrixPairState(Qureg matr, Qureg vec) {
369 
370  // Remember that for every amplitude that `vec` stores on the node,
371  // `matr` stores an entire column. Ergo there are always an integer
372  // number (in fact, a power of 2) number of `matr`s columns on each node.
373  // Since the total size of `vec` (between all nodes) is one column
374  // and each node stores (possibly) multiple columns (vec.numAmpsPerChunk as many),
375  // `vec` can be fit entirely inside a single node's matr.pairStateVec (with excess!)
376 
377  // copy this node's vec segment into this node's matr pairState (in the right spot)
378  long long int numLocalAmps = vec.numAmpsPerChunk;
379  long long int myOffset = vec.chunkId * numLocalAmps;
380  memcpy(&matr.pairStateVec.real[myOffset], vec.stateVec.real, numLocalAmps * sizeof(qreal));
381  memcpy(&matr.pairStateVec.imag[myOffset], vec.stateVec.imag, numLocalAmps * sizeof(qreal));
382 
383  // we now want to share this node's vec segment with other node, so that
384  // vec is cloned in every node's matr.pairStateVec
385 
386  // work out how many messages needed to send vec chunks (2GB limit)
387  long long int maxMsgSize = MPI_MAX_AMPS_IN_MSG;
388  if (numLocalAmps < maxMsgSize)
389  maxMsgSize = numLocalAmps;
390  // safely assume MPI_MAX... = 2^n, so division always exact:
391  int numMsgs = numLocalAmps / maxMsgSize;
392 
393  // every node gets a turn at being the broadcaster
394  for (int broadcaster=0; broadcaster < vec.numChunks; broadcaster++) {
395 
396  long long int otherOffset = broadcaster * numLocalAmps;
397 
398  // every node sends a slice of qureg's pairState to every other
399  for (int i=0; i< numMsgs; i++) {
400 
401  // by sending that slice in further slices (due to bandwidth limit)
402  MPI_Bcast(
403  &matr.pairStateVec.real[otherOffset + i*maxMsgSize],
404  maxMsgSize, MPI_QuEST_REAL, broadcaster, MPI_COMM_WORLD);
405  MPI_Bcast(
406  &matr.pairStateVec.imag[otherOffset + i*maxMsgSize],
407  maxMsgSize, MPI_QuEST_REAL, broadcaster, MPI_COMM_WORLD);
408  }
409  }
410 }

542 void compressPairVectorForSingleQubitDepolarise(Qureg qureg, const int targetQubit){
543  long long int sizeInnerBlock, sizeInnerHalfBlock;
544  long long int sizeOuterColumn, sizeOuterHalfColumn;
545  long long int thisInnerBlock, // current block
546  thisOuterColumn, // current column in density matrix
547  thisIndex, // current index in (density matrix representation) state vector
548  thisIndexInOuterColumn,
549  thisIndexInInnerBlock;
550 
551  int outerBit;
552 
553  long long int thisTask;
554  const long long int numTasks=qureg.numAmpsPerChunk>>1;
555 
556  // set dimensions
557  sizeInnerHalfBlock = 1LL << targetQubit;
558  sizeInnerBlock = 2LL * sizeInnerHalfBlock;
559  sizeOuterHalfColumn = 1LL << qureg.numQubitsRepresented;
560  sizeOuterColumn = 2LL * sizeOuterHalfColumn;
561 
562 # ifdef _OPENMP
563 # pragma omp parallel \
564  default (none) \
565  shared (sizeInnerBlock,sizeInnerHalfBlock,sizeOuterColumn,sizeOuterHalfColumn,qureg) \
566  private (thisTask,thisInnerBlock,thisOuterColumn,thisIndex,thisIndexInOuterColumn, \
567  thisIndexInInnerBlock,outerBit)
568 # endif
569  {
570 # ifdef _OPENMP
571 # pragma omp for schedule (static)
572 # endif
573  // thisTask iterates over half the elements in this process' chunk of the density matrix
574  // treat this as iterating over all columns, then iterating over half the values
575  // within one column.
576  // If this function has been called, this process' chunk contains half an
577  // outer block or less
578  for (thisTask=0; thisTask<numTasks; thisTask++) {
579  // we want to process all columns in the density matrix,
580  // updating the values for half of each column (one half of each inner block)
581  thisOuterColumn = thisTask / sizeOuterHalfColumn;
582  thisIndexInOuterColumn = thisTask&(sizeOuterHalfColumn-1); // thisTask % sizeOuterHalfColumn
583  thisInnerBlock = thisIndexInOuterColumn/sizeInnerHalfBlock;
584  // get index in state vector corresponding to upper inner block
585  thisIndexInInnerBlock = thisTask&(sizeInnerHalfBlock-1); // thisTask % sizeInnerHalfBlock
586  thisIndex = thisOuterColumn*sizeOuterColumn + thisInnerBlock*sizeInnerBlock
587  + thisIndexInInnerBlock;
588  // check if we are in the upper or lower half of an outer block
589  outerBit = extractBit(targetQubit, (thisIndex+qureg.numAmpsPerChunk*qureg.chunkId)>>qureg.numQubitsRepresented);
590  // if we are in the lower half of an outer block, shift to be in the lower half
591  // of the inner block as well (we want to dephase |0><0| and |1><1| only)
592  thisIndex += outerBit*(sizeInnerHalfBlock);
593

595  // dephase in the chunk of the state vector on this process, in the
596  // density matrix representation.
597  // thisTask is the index of the pair element in pairStateVec
598  // we will populate the second half of pairStateVec with this process'
599  // data to send
600 
601  qureg.pairStateVec.real[thisTask+numTasks] = qureg.stateVec.real[thisIndex];
602  qureg.pairStateVec.imag[thisTask+numTasks] = qureg.stateVec.imag[thisIndex];
603 
604  }
605  }

630  sizeInnerQuarterBlockQ2 = sizeInnerHalfBlockQ2 >> 1;
631  sizeInnerBlockQ2 = sizeInnerHalfBlockQ2 << 1;
632  sizeInnerBlockQ1 = 2LL * sizeInnerHalfBlockQ1;
633  sizeOuterColumn = 1LL << qureg.numQubitsRepresented;
634  sizeOuterQuarterColumn = sizeOuterColumn >> 2;
635 
636 # ifdef _OPENMP
637 # pragma omp parallel \
638  default (none) \
639  shared (sizeInnerBlockQ1,sizeInnerHalfBlockQ1,sizeInnerQuarterBlockQ2,sizeInnerHalfBlockQ2,sizeInnerBlockQ2, \
640  sizeOuterColumn, \
641  sizeOuterQuarterColumn,qureg) \
642  private (thisTask,thisInnerBlockQ2,thisOuterColumn,thisIndex,thisIndexInOuterColumn, \
643  thisIndexInInnerBlockQ1,thisIndexInInnerBlockQ2,thisInnerBlockQ1InInnerBlockQ2,outerBitQ1,outerBitQ2)
644 # endif
645  {
646 # ifdef _OPENMP
647 # pragma omp for schedule (static)
648 # endif
649  // thisTask iterates over half the elements in this process' chunk of the density matrix
650  // treat this as iterating over all columns, then iterating over half the values
651  // within one column.
652  // If this function has been called, this process' chunk contains half an
653  // outer block or less
654  for (thisTask=0; thisTask<numTasks; thisTask++) {
655  // we want to process all columns in the density matrix,
656  // updating the values for half of each column (one half of each inner block)
657  thisOuterColumn = thisTask / sizeOuterQuarterColumn;
658  // thisTask % sizeOuterQuarterColumn
659  thisIndexInOuterColumn = thisTask&(sizeOuterQuarterColumn-1);
660  thisInnerBlockQ2 = thisIndexInOuterColumn / sizeInnerQuarterBlockQ2;
661  // thisTask % sizeInnerQuarterBlockQ2;
662  thisIndexInInnerBlockQ2 = thisTask&(sizeInnerQuarterBlockQ2-1);
663  thisInnerBlockQ1InInnerBlockQ2 = thisIndexInInnerBlockQ2 / sizeInnerHalfBlockQ1;
664  // thisTask % sizeInnerHalfBlockQ1;
665  thisIndexInInnerBlockQ1 = thisTask&(sizeInnerHalfBlockQ1-1);
666 
667  // get index in state vector corresponding to upper inner block
668  thisIndex = thisOuterColumn*sizeOuterColumn + thisInnerBlockQ2*sizeInnerBlockQ2
669  + thisInnerBlockQ1InInnerBlockQ2*sizeInnerBlockQ1 + thisIndexInInnerBlockQ1;
670 
671  // check if we are in the upper or lower half of an outer block for Q1
672  outerBitQ1 = extractBit(targetQubit, (thisIndex+qureg.numAmpsPerChunk*qureg.chunkId)>>qureg.numQubitsRepresented);
673  // if we are in the lower half of an outer block, shift to be in the lower half
674  // of the inner block as well (we want to dephase |0><0| and |1><1| only)
675  thisIndex += outerBitQ1*(sizeInnerHalfBlockQ1);
676 
677  // check if we are in the upper or lower half of an outer block for Q2
678  outerBitQ2 = extractBit(qubit2, (thisIndex+qureg.numAmpsPerChunk*qureg.chunkId)>>qureg.numQubitsRepresented);
679  // if we are in the lower half of an outer block, shift to be in the lower half
680  // of the inner block as well (we want to dephase |0><0| and |1><1| only)
681  thisIndex += outerBitQ2*(sizeInnerQuarterBlockQ2<<1);
682 
683  // NOTE: at this point thisIndex should be the index of the element we want to
684  // dephase in the chunk of the state vector on this process, in the
685  // density matrix representation.
686  // thisTask is the index of the pair element in pairStateVec
687 
688  // state[thisIndex] = (1-depolLevel)*state[thisIndex] + depolLevel*(state[thisIndex]
689  // + pair[thisTask])/2
690  qureg.pairStateVec.real[thisTask+numTasks*2] = qureg.stateVec.real[thisIndex];
691  qureg.pairStateVec.imag[thisTask+numTasks*2] = qureg.stateVec.imag[thisIndex];
692  }
693  }
