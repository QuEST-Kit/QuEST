110 int isMatrixUnitary(ComplexMatrix2 u) {
111  if ( absReal( u.r0c0.real*u.r0c0.real
112  + u.r0c0.imag*u.r0c0.imag
113  + u.r1c0.real*u.r1c0.real
114  + u.r1c0.imag*u.r1c0.imag - 1) > REAL_EPS ) return 0;
115  if ( absReal( u.r0c1.real*u.r0c1.real
116  + u.r0c1.imag*u.r0c1.imag
117  + u.r1c1.real*u.r1c1.real
118  + u.r1c1.imag*u.r1c1.imag - 1) > REAL_EPS ) return 0;
119  if ( absReal( u.r0c0.real*u.r0c1.real
120  + u.r0c0.imag*u.r0c1.imag
121  + u.r1c0.real*u.r1c1.real
122  + u.r1c0.imag*u.r1c1.imag) > REAL_EPS ) return 0;
123  if ( absReal( u.r0c1.real*u.r0c0.imag
124  - u.r0c0.real*u.r0c1.imag
125  + u.r1c1.real*u.r1c0.imag
126  - u.r1c0.real*u.r1c1.imag) > REAL_EPS ) return 0;
127  return 1;
128 }
