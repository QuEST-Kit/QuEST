183 int validateMatrixIsUnitary(ComplexMatrix2 u){
184 
185  if ( fabs(u.r0c0.real*u.r0c0.real
186  + u.r0c0.imag*u.r0c0.imag
187  + u.r1c0.real*u.r1c0.real
188  + u.r1c0.imag*u.r1c0.imag - 1) > REAL_EPS ) return 0;
189  // check
190  if ( fabs(u.r0c1.real*u.r0c1.real
191  + u.r0c1.imag*u.r0c1.imag
192  + u.r1c1.real*u.r1c1.real
193  + u.r1c1.imag*u.r1c1.imag - 1) > REAL_EPS ) return 0;
194 
195  if ( fabs(u.r0c0.real*u.r0c1.real
196  + u.r0c0.imag*u.r0c1.imag
197  + u.r1c0.real*u.r1c1.real
198  + u.r1c0.imag*u.r1c1.imag) > REAL_EPS ) return 0;
199 
200  if ( fabs(u.r0c1.real*u.r0c0.imag
201  - u.r0c0.real*u.r0c1.imag
202  + u.r1c1.real*u.r1c0.imag
203  - u.r1c0.real*u.r1c1.imag) > REAL_EPS ) return 0;
204 
205  return 1;
